+'use strict';

+const test = require('node:test');
+const assert = require('node:assert/strict');
+const fs = require('fs');
+const path = require('path');
+const vm = require('vm');
+
+// extractHtmlBetweenMarkers and the HTML_*_MARKER constants are not exported
+// from eagle-harness.js. We extract them from source and run in a vm sandbox
+// so these tests exercise the real implementation byte-for-byte without
+// modifying the harness export surface (out of scope for this stage).
+const HARNESS_SRC = fs.readFileSync(
+  path.join(__dirname, '..', 'eagle', 'eagle-harness.js'),
+  'utf8'
+);
+
+function loadHtmlHelpers() {
+  const startMatch = HARNESS_SRC.match(
+    /const HTML_START_MARKER\s*=\s*(['"`])(.+?)\1\s*;/
+  );
+  const endMatch = HARNESS_SRC.match(
+    /const HTML_END_MARKER\s*=\s*(['"`])(.+?)\1\s*;/
+  );
+  const fnMatch = HARNESS_SRC.match(
+    /function extractHtmlBetweenMarkers\(text\)\s*\{[\s\S]*?\n\}/
+  );
+  if (!startMatch || !endMatch || !fnMatch) {
+    throw new Error(
+      'Could not extract HTML helpers from eagle-harness.js source'
+    );
+  }
+  const code = [
+    `const HTML_START_MARKER = ${JSON.stringify(startMatch[2])};`,
+    `const HTML_END_MARKER = ${JSON.stringify(endMatch[2])};`,
+    fnMatch[0],
+    '({ extractHtmlBetweenMarkers, HTML_START_MARKER, HTML_END_MARKER });',
+  ].join('\n');
+  return vm.runInNewContext(code);
+}
+
+const { extractHtmlBetweenMarkers, HTML_START_MARKER, HTML_END_MARKER } =
+  loadHtmlHelpers();
+
+test('eagle-harness HTML markers match the agreed contract strings', () => {
+  assert.equal(HTML_START_MARKER, '<<<HTML_PREVIEW_START>>>');
+  assert.equal(HTML_END_MARKER, '<<<HTML_PREVIEW_END>>>');
+});
+
+test('extractHtmlBetweenMarkers — happy path returns trimmed body', () => {
+  const body = '<!doctype html><html><body>hi</body></html>';
+  const text = `${HTML_START_MARKER}\n${body}\n${HTML_END_MARKER}`;
+  assert.equal(extractHtmlBetweenMarkers(text), body);
+});
+
+test('extractHtmlBetweenMarkers — missing start marker returns null', () => {
